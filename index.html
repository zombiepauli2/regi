function generateChattyRegex() { const word = document.getElementById('wordInput').value.trim().toLowerCase(); if (!word) { document.getElementById('regexOutput').textContent = "Bitte ein Wort eingeben!"; return null; } const minChars = 4; // Erst ab diesen Buchstaben matchen let parts = []; for (let i = 0; i < word.length; i++) { let ch = word[i]; let variants = ch; // Typische Chatty-Varianten if (ch === 'o') variants = 'oö0'; if (ch === 'i') variants = 'i|y|e|1'; if (ch === 'a') variants = 'a4'; if (ch === 'e') variants = 'e3'; if (i < minChars) { // die ersten minChars Buchstaben müssen matchen parts.push(`([${variants}][\\W_]*?)+`); } else { // Rest optional parts.push(`([${variants}][\\W_]*?)*`); } } const regexStr = "\\b" + parts.join('') + "\\b"; document.getElementById('regexOutput').textContent = regexStr; try { return new RegExp(regexStr, "i"); } catch (e) { document.getElementById('regexOutput').textContent = "Ungültige Regex: " + e.message; return null; } }
