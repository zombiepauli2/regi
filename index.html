function generateChattyRegex() { let word = document.getElementById('wordInput').value.trim(); if (!word) { document.getElementById('regexOutput').textContent = "Bitte ein Wort eingeben!"; return; } let parts = []; let first = true; for (let ch of word) { let variants = ch; if (ch.toLowerCase() === 'o') variants = 'o√∂0'; if (ch.toLowerCase() === 'i') variants = 'i|y|e|1'; if (ch.toLowerCase() === 'a') variants = 'a4'; if (ch.toLowerCase() === 'e') variants = 'e3'; if (first) { // erstes Zeichen muss sein parts.push("([" + variants + "][\\W_]*?)+"); first = false; } else { // restliche Buchstaben optional parts.push("([" + variants + "][\\W_]*?)*"); } } let regex = "\\b" + parts.join('') + "\\b"; document.getElementById('regexOutput').textContent = regex; return new RegExp(regex, "i"); }
